=====================================
  AgentServer - Unreal Engine 연동 예시
  (Lobby DS 풀 + Game DS 2단 구조)
=====================================


■ 전체 흐름
--------------------------------------
[Client UE]                  [AgentServer C#]           [Lobby DS Pool]        [Game DS (UE)]
    |                              |                    [Lobby_1 :7778]              |
    |                              |                    [Lobby_2 :7779]              |
    |                              |                    [Lobby_N :자동]              |
    |                              |                         |                      |
    +-- 1. WebSocket 연결 --------→|                         |                      |
    +-- 2. SendLogin -------------→|                         |                      |
    |                              +-- 인증/검증              |                      |
    |                              +-- 여유 로비 배정          |                      |
    |                              |   (가득차면 새 로비 생성) |                      |
    |←- 3. login_result -----------|                         |                      |
    |      (lobbyIp, lobbyPort,    |                         |                      |
    |       lobbyId)               |                         |                      |
    |                              |                         |                      |
    +-- 4. ClientTravel -----------------------------------→| (배정된 로비 접속)    |
    |      로비 월드에서 플레이     |                         |                      |
    |                              |                         |                      |
    +-- 5. SendCreateRoom --------→| → Room DB 저장          |                      |
    +-- 6. SendStartGame ---------→| → Game DS 실행 ------→  |                  →  | (프로세스 시작)
    |←- 7. game_started -----------|                         |                      |
    |      (gameIp, gamePort)      |                         |                      |
    |                              |                         |                      |
    +-- 8. ClientTravel -------------------------------------------------------→   | (게임 DS 접속)
    |      게임 플레이 (UE Replication)                                              |


■ 로비 자동 확장 동작
--------------------------------------
1. AgentServer 시작 -> Lobby_1 자동 실행 (Port:7778)
2. 유저 로그인 -> Lobby_1에 배정 (인원 적은 로비 우선)
3. Lobby_1 인원이 MaxPlayers(100)에 도달
4. 다음 유저 로그인 -> Lobby_2 자동 생성 (Port:7779) -> 배정
5. 유저 로그아웃 시 해당 로비 인원 차감
6. http://localhost:5000/api/lobbies 에서 로비 현황 실시간 조회

UAgentServerComponent::UAgentServerComponent()
{
    PrimaryComponentTick.bCanEverTick = false;
}

void UAgentServerComponent::BeginPlay()
{
    Super::BeginPlay();
    FModuleManager::Get().LoadModuleChecked("WebSockets");
}

void UAgentServerComponent::EndPlay(const EEndPlayReason::Type EndPlayReason)
{
    Disconnect();
    Super::EndPlay(EndPlayReason);
}

// ── 연결 ──
void UAgentServerComponent::Connect(const FString& Url)
{
    if (WebSocket && WebSocket->IsConnected())
    {
        Disconnect();
    }

    WebSocket = FWebSocketsModule::Get().CreateWebSocket(Url, TEXT("ws"));

    WebSocket->OnConnected().AddUObject(this, &UAgentServerComponent::OnWsConnected);
    WebSocket->OnConnectionError().AddUObject(this, &UAgentServerComponent::OnWsConnectionError);
    WebSocket->OnClosed().AddUObject(this, &UAgentServerComponent::OnWsClosed);
    WebSocket->OnMessage().AddUObject(this, &UAgentServerComponent::OnWsMessage);

    WebSocket->Connect();
}

void UAgentServerComponent::Disconnect()
{
    if (WebSocket)
    {
        if (WebSocket->IsConnected())
        {
            WebSocket->Close();
        }
        WebSocket = nullptr;
    }
}

bool UAgentServerComponent::IsConnected() const
{
    return WebSocket && WebSocket->IsConnected();
}

// ── 콜백 ──
void UAgentServerComponent::OnWsConnected()

    UE_LOG(LogTemp, Log, TEXT("[AgentServer] WebSocket Connected"));

void UAgentServerComponent::OnWsMessage(const FString& Message)
{
    TSharedPtr<FJsonObject> Json;
    auto Reader = TJsonReaderFactory<>::Create(Message);
    if (!FJsonSerializer::Deserialize(Reader, Json) || !Json.IsValid())
        return;

    FString Type = Json->GetStringField(TEXT("type"));

    // ── login_result: 배정된 로비 DS 접속 정보 포함 ──
    if (Type == TEXT("login_result"))
    {
        bool bSuccess = Json->GetBoolField(TEXT("success"));
        FString UserId = Json->HasField(TEXT("userId")) ? Json->GetStringField(TEXT("userId")) : TEXT("");
        FString LobbyIp = Json->HasField(TEXT("lobbyServerIp")) ? Json->GetStringField(TEXT("lobbyServerIp")) : TEXT("");
        int32 LobbyPort = Json->HasField(TEXT("lobbyServerPort")) ? Json->GetIntegerField(TEXT("lobbyServerPort")) : 0;
        FString LobbyId = Json->HasField(TEXT("lobbyId")) ? Json->GetStringField(TEXT("lobbyId")) : TEXT("");

        OnLoginResult.Broadcast(bSuccess, UserId, LobbyIp, LobbyPort);

        if (bSuccess)
        {
            UE_LOG(LogTemp, Log, TEXT("[AgentServer] Login OK -> %s at %s:%d"), *LobbyId, *LobbyIp, LobbyPort);
        }
    }
    // ── chat_message ──
    else if (Type == TEXT("chat_message"))
    {
        FString SenderName = Json->GetStringField(TEXT("senderName"));
        FString Msg = Json->GetStringField(TEXT("message"));
        FString RoomId = Json->HasField(TEXT("roomId")) ? Json->GetStringField(TEXT("roomId")) : TEXT("");
        OnChatReceived.Broadcast(SenderName, Msg, RoomId);
    }
    // ── room_list ──
    else if (Type == TEXT("room_list"))
    {
        OnRoomListUpdated.Broadcast(Message);
    }
    // ── room_created ──
    else if (Type == TEXT("room_created"))
    {
        FString RoomId = Json->GetStringField(TEXT("roomId"));
        FString RoomName = Json->GetStringField(TEXT("roomName"));
        OnRoomCreated.Broadcast(RoomId, RoomName);
    }
    // ── room_update (참가자 갱신) ──
    else if (Type == TEXT("room_update"))
    {
        OnRoomUpdated.Broadcast(Message);
    }
    // ── game_started: Game DS 접속 정보 ──
    else if (Type == TEXT("game_started"))
    {
        FString Ip = Json->GetStringField(TEXT("serverIp"));
        int32 Port = Json->GetIntegerField(TEXT("serverPort"));
        OnGameStarted.Broadcast(Ip, Port);

        UE_LOG(LogTemp, Log, TEXT("[AgentServer] Game started -> DS at %s:%d"), *Ip, Port);
    }
}

// ── Travel 헬퍼 ──
void UAgentServerComponent::TravelToLobby(const FString& LobbyIp, int32 LobbyPort)
{
    if (APlayerController* PC = UGameplayStatics::GetPlayerController(this, 0))
    {
        FString TravelURL = FString::Printf(TEXT("%s:%d"), *LobbyIp, LobbyPort);

        UE_LOG(LogTemp, Log, TEXT("[AgentServer] Traveling to Lobby DS: %s"), *TravelURL);
    }
}

// ── 패킷 전송 유틸 ──
void UAgentServerComponent::SendPacket(const FString& Type, const TSharedRef<FJsonObject>& Data)
{
    if (!WebSocket || !WebSocket->IsConnected())
        return;

    TSharedRef<FJsonObject> Packet = MakeShared<FJsonObject>();
    Packet->SetStringField(TEXT("type"), Type);
    Packet->SetObjectField(TEXT("data"), Data);

    FString Output;
    auto Writer = TJsonWriterFactory<>::Create(&Output);
    FJsonSerializer::Serialize(Packet, Writer);

    WebSocket->Send(Output);
}

// === MyGameInstance.h ===
UCLASS()
class UMyGameInstance : public UGameInstance
{
    GENERATED_BODY()
public:
    UPROPERTY()
    UAgentServerComponent* AgentServer;

    virtual void Init() override;
    virtual void Shutdown() override;

    // 로그인 결과 핸들러
    UFUNCTION()
    void HandleLoginResult(bool bSuccess, const FString& UserId,
        const FString& LobbyIp, int32 LobbyPort);

    // 게임 시작 핸들러
    UFUNCTION()
    void HandleGameStarted(const FString& GameIp, int32 GamePort);
};


// === MyGameInstance.cpp ===
void UMyGameInstance::Init()
{
    Super::Init();

    // 1) AgentServer 컴포넌트 생성
    AgentServer = NewObject<UAgentServerComponent>(this);
    AgentServer->RegisterComponent();

    // 2) 델리게이트 바인딩
    AgentServer->OnLoginResult.AddDynamic(this, &UMyGameInstance::HandleLoginResult);
    AgentServer->OnGameStarted.AddDynamic(this, &UMyGameInstance::HandleGameStarted);

    // 3) AgentServer에 WebSocket 연결
    AgentServer->Connect(TEXT("ws://localhost:5000/ws"));
}

void UMyGameInstance::Shutdown()
{
    if (AgentServer)
    {
        AgentServer->Disconnect();
    }
    Super::Shutdown();
}

// 4) 로그인 성공 -> 로비 DS로 Travel
void UMyGameInstance::HandleLoginResult(bool bSuccess, const FString& UserId,
    const FString& LobbyIp, int32 LobbyPort)
{
    if (bSuccess)
    {
        UE_LOG(LogTemp, Log, TEXT("Login OK! Traveling to Lobby DS..."));
        AgentServer->TravelToLobby(LobbyIp, LobbyPort);
    }
    else
    {
        UE_LOG(LogTemp, Error, TEXT("Login Failed!"));
    }
}

// 5) 게임 시작 -> Game DS로 Travel
void UMyGameInstance::HandleGameStarted(const FString& GameIp, int32 GamePort)
{
    UE_LOG(LogTemp, Log, TEXT("Game started! Traveling to Game DS..."));
    AgentServer->TravelToGameServer(GameIp, GamePort);
}

void UChatTestGameInstance::OnChat(const FString& SenderName, const FString& Message,
    const FString& RoomId)
{
    // 채팅 수신 확인 로그
    UE_LOG(LogTemp, Warning, TEXT("=== CHAT RECEIVED: [%s] %s ==="), *SenderName, *Message);

    // 여기서 UI에 표시하거나, GEngine->AddOnScreenDebugMessage 로 화면에 출력
    if (GEngine)
    {
        GEngine->AddOnScreenDebugMessage(-1, 10.f, FColor::Green,
            FString::Printf(TEXT("[%s] %s"), *SenderName, *Message));
    }
}

void UChatTestGameInstance::TestSendChat(const FString& Message)
{
    if (AgentServer && AgentServer->IsConnected())
    {
        AgentServer->SendGlobalChat(Message);
        UE_LOG(LogTemp, Log, TEXT("=== SENT: %s ==="), *Message);
    }
}
