■ CPP (ChatWebSocket.cpp)
--------------------------------------
#include "ChatWebSocket.h"
#include "WebSocketsModule.h"
#include "Dom/JsonObject.h"
#include "Serialization/JsonReader.h"
#include "Serialization/JsonSerializer.h"
#include "Serialization/JsonWriter.h"

AChatWebSocket::AChatWebSocket()
{
    PrimaryActorTick.bCanEverTick = false;
}

void AChatWebSocket::BeginPlay()
{
    Super::BeginPlay();
    FModuleManager::Get().LoadModuleChecked("WebSockets");
}

void AChatWebSocket::EndPlay(const EEndPlayReason::Type EndPlayReason)
{
    Disconnect();
    Super::EndPlay(EndPlayReason);
}

// ── 연결 ──
void AChatWebSocket::Connect(const FString& Url)
{
    // ex: "ws://localhost:5000/ws"
    WebSocket = FWebSocketsModule::Get().CreateWebSocket(Url, TEXT("ws"));

    WebSocket->OnConnected().AddUObject(this, &AChatWebSocket::OnConnected);
    WebSocket->OnConnectionError().AddUObject(this, &AChatWebSocket::OnConnectionError);
    WebSocket->OnClosed().AddUObject(this, &AChatWebSocket::OnClosed);
    WebSocket->OnMessage().AddUObject(this, &AChatWebSocket::OnMessage);

    WebSocket->Connect();
}

void AChatWebSocket::Disconnect()
{
    if (WebSocket && WebSocket->IsConnected())
    {
        WebSocket->Close();
    }
    WebSocket = nullptr;
}

// ── 콜백 (서버 -> 클라이언트) ──
void AChatWebSocket::OnConnected()
{
    UE_LOG(LogTemp, Log, TEXT("WebSocket Connected"));
}

void AChatWebSocket::OnConnectionError(const FString& Error)
{
    UE_LOG(LogTemp, Error, TEXT("WebSocket Error: %s"), *Error);
}

void AChatWebSocket::OnClosed(int32 StatusCode, const FString& Reason, bool bWasClean)
{
    UE_LOG(LogTemp, Log, TEXT("WebSocket Closed: %s"), *Reason);
}

void AChatWebSocket::OnMessage(const FString& Message)
{
    TSharedPtr<FJsonObject> Json;
    auto Reader = TJsonReaderFactory<>::Create(Message);
    if (!FJsonSerializer::Deserialize(Reader, Json) || !Json.IsValid())
        return;

    FString Type = Json->GetStringField(TEXT("type"));

    if (Type == TEXT("chat_message"))
    {
        FString SenderName = Json->GetStringField(TEXT("senderName"));
        FString Msg = Json->GetStringField(TEXT("message"));
        FString RoomId = Json->GetStringField(TEXT("roomId"));
        OnChatReceived.Broadcast(SenderName, Msg, RoomId);
    }
    else if (Type == TEXT("room_list"))
    {
        OnRoomListUpdated.Broadcast(Message);
    }
    else if (Type == TEXT("game_started"))
    {
        FString Ip = Json->GetStringField(TEXT("serverIp"));
        int32 Port = Json->GetIntegerField(TEXT("serverPort"));
        OnGameStarted.Broadcast(Ip, Port);
    }
}

// ── 패킷 전송 유틸 ──
void AChatWebSocket::SendPacket(const FString& Type, const TSharedRef<FJsonObject>& Data)
{
    if (!WebSocket || !WebSocket->IsConnected())
        return;

    TSharedRef<FJsonObject> Packet = MakeShared<FJsonObject>();
    Packet->SetStringField(TEXT("type"), Type);
    Packet->SetObjectField(TEXT("data"), Data);

    FString Output;
    auto Writer = TJsonWriterFactory<>::Create(&Output);
    FJsonSerializer::Serialize(Packet, Writer);

    WebSocket->Send(Output);
}

// ── 전송 함수 ──
void AChatWebSocket::SendLogin(const FString& UserId, const FString& DisplayName)
{
    auto Data = MakeShared<FJsonObject>();
    Data->SetStringField(TEXT("userId"), UserId);
    Data->SetStringField(TEXT("displayName"), DisplayName);
    SendPacket(TEXT("login"), Data);
}

void AChatWebSocket::SendChat(const FString& Message, const FString& RoomId)
{
    auto Data = MakeShared<FJsonObject>();
    Data->SetStringField(TEXT("message"), Message);
    Data->SetStringField(TEXT("roomId"), RoomId);
    Data->SetNumberField(TEXT("type"), 1); // Room=1
    SendPacket(TEXT("chat"), Data);
}

void AChatWebSocket::SendCreateRoom(const FString& RoomName, int32 MaxPlayers, const FString& MapName)
{
    auto Data = MakeShared<FJsonObject>();
    Data->SetStringField(TEXT("roomName"), RoomName);
    Data->SetNumberField(TEXT("maxPlayers"), MaxPlayers);
    Data->SetStringField(TEXT("mapName"), MapName);
    SendPacket(TEXT("create_room"), Data);
}

void AChatWebSocket::SendJoinRoom(const FString& RoomId)
{
    auto Data = MakeShared<FJsonObject>();
    Data->SetStringField(TEXT("roomId"), RoomId);
    SendPacket(TEXT("join_room"), Data);
}

void AChatWebSocket::SendStartGame(const FString& RoomId)
{
    auto Data = MakeShared<FJsonObject>();
    Data->SetStringField(TEXT("roomId"), RoomId);
    SendPacket(TEXT("start_game"), Data);
}
